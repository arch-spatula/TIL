## n^2 배열 자르기

```js
/**
 * @param {number} n
 * @param {number} left
 * @param {number} right
 * @returns {number[]}
 */
function solution(n, left, right) {
  return squareArr(n)
    .flat()
    .slice(left, right + 1);
}

/**
 * @param {number} n
 * @returns {number[][]}
 */
export function squareArr(n) {
  const result = [];
  for (let i = 1; i <= n; i++) {
    const row = [];
    for (let j = 1; j <= n; j++) row.push(Math.max(i, j));
    result.push(row);
  }
  return result;
}

export default solution;
```

이 힌트가 상당히 유용했습니다. i, j 행렬을 표현하는데 몫과 나머지로 표현할 것이라는 생각은 못했습니다.

```js
/**
 * @param {number} n
 * @param {number} left
 * @param {number} right
 * @returns {number[]}
 */
function solution(n, left, right) {
  return Array.from({ length: right - left + 1 }, (_, idx) => {
    const [i, j] = [Math.floor((idx + left) / n) + 1, ((idx + left) % n) + 1];
    return Math.max(i, j);
  });
}

export default solution;
```
